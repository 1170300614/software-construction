/* class contents */

C++ Standard Library includes
the Standard Template Library
(STL) that provides a framework
of templates for common data
structures and algorithms.
template<typename T>
void Swap(T & a, T & b) {
T temp = b;
b = a;
a = temp;
}
string hello = “world!”
string world = “Hello,”;
Swap( world, hello );
cout << hello << world << endl;
