public class MyStaticMethods{
3.4 Object-Oriented Programming (OOP)
private String name;
private static String staticStr = "STATIC-STRING";
public MyStatic (String n){
this.name = n;
}
public static void testStaticMethod(){
System.out.println(“Hey... I am in static method...”); //you can call static variables here,but
System.out.println(MyStaticMethods.staticStr); //you can not call instance variables here!??
}
静态方法无法调用非静态成员（方法和变量）
public void testObjectMethod(){
System.out.println("Hey i am in non-static method");//you can also call static variables here
System.out.println(MyStaticMethods.staticStr); //you can call instance variables here
System.out.println("Name: "+this.name);
}
public static void main(String a[]){
//By using class name, you can call static method
MyStatic.testStaticMethod();
MyStatic msm = new MyStatic ("Java2novice");
msm.testObjectMethod();
}
}
