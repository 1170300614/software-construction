flip(){
Fromspace, Tospace = Tospace, Fromspace
top_of_space = Tospace + space_size
scan = free = Tospace
for R in Roots {R = copy(R)}
while scan < free {
for P in Children(scan) {*P = copy(*P)}

}
}
copy(P) {
if forwarded(P){return forwarding_address(P)}
else {
addr = free
move(P,free)
free = free + size(P)
forwarding_address(P) = addr
return addr
}
