6.2 Design Patterns for Maintainability
public class RandomNumberGenerator extends NumberGenerator {
private Random random = new Random(); // 随机数生成器
private int number;
// 当前数值
public int getNumber() {
return number;
}
// 获取当前数值
public void execute() {
for (int i = 0; i < 20; i++) {
number = random.nextInt(50);
notify();
}
}
}
